101int8_t convertToBcdArray (uint32_t data, uint8_t digits, uint8_t * bcd_number)
102{
103 if (digits > 10) {
104 return -1;
105 }
106 // Inicializar el array de salida a 0
107 for (uint8_t i = 0; i < digits; i++) {
108 bcd_number[i] = 0;
109 }
110 for (uint8_t i = 0; i < digits; i++) {
111 bcd_number[digits - i - 1] = data % 10; // dígito menos significativo
112 data = data / 10; // se borra el dígito menos significativo
113 }
114 // Si después de convertir hay datos restantes en el número de entrada,
115 // significa que el número de dígitos era insuficiente
116 if (data > 0) {
117 return -1;
118 }
119
120 return 0;
121}

130void BCDtoGPIO(uint8_t digit, gpioConfig_t *gpio_config) {
131 for (uint8_t i = 0; i < N_BITS; i++) {
132 GPIOInit(gpio_config[i].pin, gpio_config[i].dir);
133 }
134 for (uint8_t i = 0; i < N_BITS; i++) {
135 if ((digit & (1 << i)) == 0) { //máscara (1 << i) y operación AND para determinar el estado de cada bit
136 GPIOOff(gpio_config[i].pin); // si apaga el GPIO si el bit es 0
137 } else {
138 GPIOOn(gpio_config[i].pin); // se prende el GPIO si el bit es 1
139 }
140 }
141}

150void displayNumberOnLCD(uint32_t data, gpioConfig_t *data_gpio_config, gpioConfig_t *digit_gpio_config) {
151 uint8_t bcd_array[LCD_DIGITS];
152
153 if (convertToBcdArray(data, LCD_DIGITS, bcd_array) != 0) {
154 printf("Error: el número es demasiado grande para la cantidad de dígitos.\n");
155 return;
156 }
157
158 // inicializar los pines del LCD
159 for (uint8_t i = 0; i < LCD_DIGITS; i++) {
160 GPIOInit(digit_gpio_config[i].pin, digit_gpio_config[i].dir);
161 }
162
163 // mostrar cada dígito en el display
164 for (uint8_t i = 0; i < LCD_DIGITS; i++) {
165 for (uint8_t j = 0; j < LCD_DIGITS; j++) {
166 GPIOOff(digit_gpio_config[j].pin); // se apagan todos los dígitos
167 }
168
169 GPIOOn(digit_gpio_config[i].pin); // pulso (on-off)
170
171 BCDtoGPIO(bcd_array[i], data_gpio_config);
172 }
173}

148static void SensarRadADCTask(void *pvParameter){
149 uint16_t tension;
150
151 while (true) {
152 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
153
154 if (sistema_encendido) {
155 //lectura y conversión
156 AnalogInputReadSingle(CH1, &tension);
157 radiacion = (float)tension * (100.0 / 3300.0); // 3.3V -> 100 mR/h
158
159 char texto[50];
160
161 // radiación mayor a 40mR/h -> riesgo ambiente
162 if (radiacion > 40) {
163 printf(texto, sizeof(texto),
164 "Radiación %.1fmR/h - RADIACIÓN ELEVADA\r\n", radiacion);
165 } else {
166 printf(texto, sizeof(texto),
167 "Radiación %.1fmR/h\r\n", radiacion);
168 }
169
170 UartSendString(UART_PC, texto);
171 }
172 }

182static void LedsTask(void *pvParameter){
183 while (true) {
184 vTaskDelay(pdMS_TO_TICKS(500));
185
186 if (!sistema_encendido) {
187 GPIOOff(GPIO_1);
188 GPIOOff(GPIO_2);
189 GPIOOff(GPIO_3);
190 }
191
192 //Led verde 1 - Led amarillo 2 - Led rojo 3
193 if (humedad > 85 && temperatura >= 0 && temperatura <= 2) {
194 GPIOOn(GPIO_3);
195 GPIOff(GPIO_2);
196 GPIOOff(GPIO_1);
197 } else if (radiacion > 40) {
198 GPIOOn(GPIO_2);
199 GPIOOff(GPIO_3);
200 GPIOOff(GPIO_1);
201 } else {
202 GPIOOn(GPIO_1);
203 GPIOOff(GPIO_2);
204 GPIOOff(GPIO_3);
205 }
206 }
207}
